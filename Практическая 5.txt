FROM TEACHER

(KOD_KAFEDRU < 550 OR INDEF_KOD > 60);

 KOD_KAFEDRU + INDEF_KOD AS 'Его зарплата'
FROM TEACHER
WHERE KOD_KAFEDRU + INDEF_KOD > 3500

 KURS AS 'Курс'

FROM TEACHER

'%o%o%o%o%o%';
